Integration Code

<!-- Include the Logger Simple script -->
<script src="https://connexion.logger-simple.com/api/script.js"></script>

<!-- Your login button container -->
<div id="login-button"></div>

<script>
// Initialize Logger Simple
LoggerSimple.init({
    app_id: 'your_app_id_here',
    redirect_uri: 'https://yoursite.com/callback',
    onSuccess: function(user) {
        console.log('User logged in:', user);
        // Handle successful login
        document.getElementById('welcome').innerHTML = 
            'Welcome, ' + user.username + '!';
    },
    onError: function(error) {
        console.error('Login error:', error);
        alert('Login failed: ' + error.description);
    }
});

// Render login button
LoggerSimple.renderButton('#login-button', {
    text: 'Login with Logger Simple',
    style: 'primary',
    size: 'medium'
});
</script>
